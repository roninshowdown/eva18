#define _POSIX_C_SOURCE 200809L // <> Anforderung der POSIX-Version
#define _XOPEN_SOURCE 700 // <> Anforderung  der XSI-Version
#ifdef __cplusplus
extern "C" {
#endif
#if defined(_WIN32) || defined(_WIN64) // <> Microsoft-Windows ist ohne weiteres nicht POSIX-kompatibel
# error "Not a POSIX system! Compilation not possible."
#endif
#include <unistd.h>
#include <limits.h>
#include <stdlib.h>
#include <stdio.h>
#include <assert.h>
#ifdef __cplusplus
}
#endif
const size_t innerSize = 4096;
// const size_t outerSize = 4096 * 1024;
const size_t outerSize = 4096 * 100;
unsigned char array[outerSize][innerSize] = {};
int main()
{
#ifdef PAGESIZE
    const long pageSize = PAGESIZE;
#else
    const long pageSize = sysconf(_SC_PAGE_SIZE);
#endif
    printf("pagesize: %ld\n", pageSize);
    for(size_t k = 0; k < innerSize; k += 1)
    {
        for(size_t i = 0; i < outerSize; i += 1)
        {
            array[i][k] = static_cast<unsigned char>((i * k) % 256);
        }
    }
    exit(EXIT_SUCCESS);
}
